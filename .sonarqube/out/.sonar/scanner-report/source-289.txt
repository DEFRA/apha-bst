CREATE OR ALTER   PROCEDURE [dbo].[sp_Site_Trainee_Get] --'110938000'
@PlantNo NVarChar(11)
AS
BEGIN
SELECT DISTINCT    
	P.PersonID, 
	P.Person
	
FROM  tSites S INNER JOIN
      tPerson P ON S.PlantNo = P.LocationID
	 
WHERE     (S.PlantNo = @PlantNo)
END
